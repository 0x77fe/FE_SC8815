1. 芯片概述

SC8815 是 Southchip 出品的高效率同步双向 Buck-Boost 充电控制器，集成 NVDC 电源路径管理，支持：

  • 充电模式：涓流充电 → 恒流充电 (CC) → 恒压充电 (CV) → 充电终止
  • 放电模式 (OTG)：电池反向升/降压输出可调 VBUS 电压
  • I²C 可编程：充电电流/电压、放电输出电压、输入/输出电流限制、开关频率
  • 10-bit ADC 监控：VBUS 电压 / VBAT 电压 / IBUS 电流 / IBAT 电流 / ADIN 外部电压
  • 完善保护：UVP 输入欠压、OVP 过压、OCP 过流、SCP 短路、OTP 过温 (165°C)

2. 快速开始

#include <Wire.h>
#include "sc8815.h"

SC8815 sc;

void setup() {
    Serial.begin(115200);
    Wire.begin(21, 22);        // ESP32: SDA=21, SCL=22
    Wire.setClock(400000);     // 400kHz I²C

    if (!sc.begin()) {
        Serial.println("SC8815 未检测到!");
        return;
    }

    // === 充电配置 ===
    // 3S 电池, 4.20V/节 (内部设定, IRCOMP=0mΩ)
    SC8815_VBATSet_Config_t bat_config;
    bat_config.ircomp = SC8815_IRCOMP_0mOhm;
    bat_config.sel    = SC8815_VBAT_SEL_INTERNAL;
    bat_config.csel   = SC8815_CSEL_3S;
    bat_config.vcell  = SC8815_VCELL_4V20;
    sc.setVBATSet(bat_config);

    // 电流限制: IBUS=2A, IBAT=2A
    sc.setIBUSLimit(2.0);   // 参数为安培 (A)
    sc.setIBATLimit(2.0);

    // 充电控制: 内部 FB, 启用涓流, 启用自动终止, 启用 OVP
    SC8815_CTRL1_Config_t ctrl1_config;
    ctrl1_config.ichar_sel   = SC8815_ICHAR_SEL_IBUS;
    ctrl1_config.dis_trickle = SC8815_DIS_TRICKLE_DISABLE;   // 启用涓流充电
    ctrl1_config.dis_term    = SC8815_DIS_TERM_DISABLE;      // 启用自动充电终止
    ctrl1_config.fb_sel      = SC8815_FB_SEL_INTERNAL;
    ctrl1_config.trickle_set = SC8815_TRICKLE_SET_70;
    ctrl1_config.dis_ovp     = SC8815_DIS_OVP_DISABLE;       // 启用过压保护
    sc.setCTRL1(ctrl1_config);
}

void loop() {
    // 读取 VBUS / VBAT 电压和电流
    uint16_t vbus_mV, vbat_mV;
    double   ibus_A, ibat_A;
    sc.readVBUSVoltage(vbus_mV);
    sc.readVBATVoltage(vbat_mV);
    sc.readIBUSCurrent(ibus_A);
    sc.readIBATCurrent(ibat_A);
    Serial.printf("VBUS:%umV, VBAT:%umV, IBUS:%.2fA, IBAT:%.2fA\n",
                  vbus_mV, vbat_mV, ibus_A, ibat_A);
    delay(1000);
}
提示：调用 begin() 前必须自行初始化 I²C 总线（Wire.begin / Wire.setPins / Wire.setClock）。库不会自动重置 Wire 配置。

3. I²C 通信

参数值
7-bit 地址0x74
8-bit 写地址0xE8 (0x74 << 1)
8-bit 读地址0xE9 (0x74 << 1 | 1)
支持速率100kHz (标准) / 400kHz (快速)
寄存器数量25 个 (地址 0x00~0x19，0x18 保留)
通信流程：写操作 → 发送 START + 设备地址/W + 寄存器地址 + 数据 + STOP；读操作 → 发送 START + 设备地址/W + 寄存器地址 + RESTART + 设备地址/R + 读数据 + STOP。

5.27 DIS_ShortFoldBack — 短路电流折返 CTRL3[2]

6.4 VBUSREF_I — 内部 VBUS 参考电压 0x01+0x02

RW setVBUSRefI
bool setVBUSRefI(uint16_t ref_mV);

设置内部 VBUS 参考电压 (10-bit)。

范围：2~2048 mV。

公式：VBUSREF_I = (4 × VHI + VLO + 1) × 2mV

输出：VBUS = VBUSREF_I × VBUS_RATIO

仅在 FB_SEL=0 (内部设定) 时生效。写入前请确认 CTRL1 的 FB_SEL=0。
读取当前 VBUSREF_I 值可使用 readRegister(SC8815_REG::VBUSREF_I_SET, vhi) 和 readRegister(SC8815_REG::VBUSREF_I_SET2, vlo_reg)，然后按公式计算。

6.5 VBUSREF_E — 外部 FB 参考电压 0x03+0x04

RW setVBUSRefE
bool setVBUSRefE(uint16_t ref_mV);

设置外部 FB 引脚参考电压 (10-bit)。

推荐范围：0.5V ~ 2.048V。

输出：VBUS = VBUSREF_E × (1 + RUP / RDOWN)

仅在 FB_SEL=1 (外部电阻设定) 时生效。FB_SEL=1 时仍需配置 VBUS_RATIO（仅影响 ADC 监控量程）。
读取当前 VBUSREF_E 值可使用 readRegister(SC8815_REG::VBUSREF_E_SET, vhi) 和 readRegister(SC8815_REG::VBUSREF_E_SET2, vlo_reg)，然后按公式计算。

6.6 IBUS_LIM — IBUS 电流限制 0x05

RW setIBUSLimit
bool setIBUSLimit(double current_A);

设置 IBUS 电流限制（充/放电模式均生效）。参数 current_A 单位为安培 (A)。

公式：IBUS_LIM(A) = (IBUS_LIM_SET + 1) / 256 × IBUS_RATIO × 10mΩ / RS1

自动读取当前 IBUS_RATIO 和 RS1=10mΩ 计算寄存器值。

实际限流精度取决于采样电阻温度系数。最小值不低于 0.3A（数据手册要求）。
读取当前 IBUS_LIM 设定值可使用 readRegister(SC8815_REG::IBUS_LIM_SET, value)，然后按公式反算电流值。

6.7 IBAT_LIM — IBAT 电流限制 0x06

RW setIBATLimit
bool setIBATLimit(double current_A);

设置 IBAT 电流限制（充/放电模式均生效）。参数 current_A 单位为安培 (A)。

公式：IBAT_LIM(A) = (IBAT_LIM_SET + 1) / 256 × IBAT_RATIO × 10mΩ / RS2

读取当前 IBAT_LIM 设定值可使用 readRegister(SC8815_REG::IBAT_LIM_SET, value)，然后按公式反算电流值。

6.8 VINREG — 动态电源管理 0x07

RW setVINREG
bool setVINREG(uint16_t voltage_mV);

设置 VINREG 参考电压（充电模式动态电源管理）。

公式：VINREG = (VINREG_SET + 1) × VINREG_RATIO (mV)

自动读取 CTRL0 中的 VINREG_RATIO 位确定步进值 (40mV 或 100mV)。

当 VBUS 电压降至 VINREG 时，自动减小充电电流以保护适配器。

读取当前 VINREG 设定值可使用 readRegister(SC8815_REG::VINREG_SET, value)，然后按公式反算电压值。

    ctrl3_config.en_pfm             = SC8815_EN_PFM_DISABLE;

}

void loop() {
    uint16_t vbus_mV, vbat_mV;
    double   ibus_A, ibat_A;
    sc.readVBUSVoltage(vbus_mV);
    sc.readVBATVoltage(vbat_mV);
    sc.readIBUSCurrent(ibus_A);
    sc.readIBATCurrent(ibat_A);

    Serial.printf("VBUS=%umV  VBAT=%umV  IBUS=%.2fA  IBAT=%.2fA\n",
                  vbus_mV, vbat_mV, ibus_A, ibat_A);

    SC8815_Status_t status;
    sc.getStatus(status);
    if (status.ACOK) Serial.println("  → 适配器已插入");
    if (status.EOC)  Serial.println("  → 充电已终止");

    delay(1000);
}

8. 注意事项

8.1 寄存器操作

  • Read-Modify-Write：所有 setCTRL0()、setCTRL1()、setCTRL2()、setRatio()、setIRQMask() 均采用 RMW 策略保护寄存器中的 Reserved bits，避免因直接覆写导致芯片异常。
  • CTRL2 FACTORY 位：数据手册强制要求 MCU 上电后将 CTRL2[3] 置 1。库中 setCTRL2() 默认参数已自动设为 SC8815_FACTORY_POWERUP。
  • PSTOP 限制：CTRL0、CTRL1、CTRL2、CTRL3 寄存器建议仅在 PSTOP=HIGH (待机状态) 时修改。

8.2 I²C 初始化

  • begin() 不会调用 Wire.begin()。调用方必须在调用 begin() 之前自行完成 I²C 总线的引脚配置和时钟设置。
  • 推荐 I²C 速率 400kHz (Fast Mode)。

8.3 ADC 使用

  • 读取任何 ADC 值之前，必须先通过 setCTRL3() 将 ad_start 设为 SC8815_AD_START_ENABLE。
  • ADC 为 10-bit 分辨率，分布在两个寄存器中 (高 8 位 + 低 2 位)。
  • VBUS_RATIO=12.5x 时最大可测 ~25.6V，超出将饱和。

8.4 电流限制假设

  • 所有电流计算默认假设采样电阻 RSNS1 = RSNS2 = 10mΩ。如果实际电路使用不同阻值的采样电阻，需要修改库代码中的对应常量。
  • 实际限流精度取决于采样电阻的温度系数。

8.5 保护功能

保护触发条件行为
OVP(BAT)电池电压超过阈值停止充电 / 关断输出
OVP(BUS)输出电压超过阈值关断输出 (可通过 DIS_OVP 控制)
OCP电流超过 IBUS_LIM / IBAT_LIM限流
SCPVBUS < 1V (短路)电流折返至 22%(IBUS) / 10%(IBAT)
OTPTj > 165°C关断输出, 滞后 ~35°C 恢复

8.6 寄存器缓存

  • 库内部维护 6 个缓存变量（_cache_ratio、_cache_ctrl0、_cache_ctrl1、_cache_ctrl2、_cache_ctrl3、_cache_mask），在构造时使用数据手册的 POR 默认值初始化。
  • Setter 更新缓存：所有 set*() 函数（包括 setCTRL3、setPGATE、setGPO）在成功写入硬件后同步更新对应缓存变量。
  • 缓存同步：所有 set*() 函数在成功写入硬件后自动更新对应缓存。如需从硬件重新同步缓存值，可使用 readRegister() 直接读取对应寄存器（如 SC8815_REG::RATIO、SC8815_REG::CTRL0_SET、SC8815_REG::CTRL3_SET 等），手动解析位字段后更新缓存。
  • 缓存作为只读源：电流限制计算（setIBUSLimit、setIBATLimit）、VINREG 计算（setVINREG）和 ADC 读数均从缓存读取比率和配置位，不发出额外的 I²C 请求。这意味着如果缓存与实际硬件状态不同步，计算结果将偏离实际。可通过 readRegister() 直接读取对应寄存器来验证硬件状态。
  • CTRL3 缓存：CTRL3_SET 寄存器虽无 reserved bits，但 setCTRL3()、setPGATE()、setGPO() 均采用 read-modify-write 策略更新 _cache_ctrl3，确保快捷函数只修改目标位而不影响其他 CTRL3 配置。
