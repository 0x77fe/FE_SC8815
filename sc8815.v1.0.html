1. 芯片概述

SC8815 是 Southchip 出品的高效率同步双向 Buck-Boost 充电控制器，集成 NVDC 电源路径管理，支持：

  • 充电模式：涓流充电 → 恒流充电 (CC) → 恒压充电 (CV) → 充电终止
  • 放电模式 (OTG)：电池反向升/降压输出可调 VBUS 电压
  • I²C 可编程：充电电流/电压、放电输出电压、输入/输出电流限制、开关频率
  • 10-bit ADC 监控：VBUS 电压 / VBAT 电压 / IBUS 电流 / IBAT 电流 / ADIN 外部电压
  • 完善保护：UVP 输入欠压、OVP 过压、OCP 过流、SCP 短路、OTP 过温 (165°C)

应用场景

场景说明
笔记本 / 平板电池充电1~4S 锂电池组，I²C 设置充电参数
移动电源 / 储能电站双向充放电，OTG 输出可调电压
USB-C PD 充电器配合 CC 控制器实现 PD 电压调节
工业电池管理系统 (BMS)12V/24V 电池组的充放电管理

2. 快速开始

#include <Wire.h>
#include "sc8815.h"

SC8815 sc;

void setup() {
    Serial.begin(115200);
    Wire.begin(21, 22);        // ESP32: SDA=21, SCL=22
    Wire.setClock(400000);     // 400kHz I²C

    if (!sc.begin()) {
        Serial.println("SC8815 未检测到!");
        return;
    }

    // 配置充电: 3S 电池, 4.20V/节, IBUS=2A, IBAT=2A
    sc.setChargeVoltage(3, SC8815_VCELL_4V20);
    sc.setChargeCurrent(2000, 2000);
    sc.setCTRL1(SC8815_ICHAR_SEL_IBUS,
                SC8815_DIS_TRICKLE_ENABLE,
                SC8815_DIS_TERM_ENABLE,
                SC8815_FB_SEL_INTERNAL,
                SC8815_TRICKLE_SET_70,
                SC8815_DIS_OVP_DISABLE);
}

void loop() {
    // 读取 VBUS / VBAT 电压和电流
    uint16_t vbus, vbat, ibus, ibat;
    sc.readVBUSVoltage(&vbus);
    sc.readVBATVoltage(&vbat);
    sc.readIBUSCurrent(&ibus);
    sc.readIBATCurrent(&ibat);
    Serial.printf("VBUS:%dmV, VBAT:%dmV, IBUS:%dmA, IBAT:%dmA\n",
                  vbus, vbat, ibus, ibat);
    delay(1000);
}
提示：调用 begin() 前必须自行初始化 I²C 总线（Wire.begin / Wire.setPins / Wire.setClock）。库不会自动重置 Wire 配置。

3. I²C 通信

参数值
7-bit 地址0x74
8-bit 写地址0xE8 (0x74 << 1)
8-bit 读地址0xE9 (0x74 << 1 | 1)
支持速率100kHz (标准) / 400kHz (快速)
寄存器数量26 个 (地址 0x00~0x1B，0x18 保留)
通信流程：写操作 → 发送 START + 设备地址/W + 寄存器地址 + 数据 + STOP；读操作 → 发送 START + 设备地址/W + 寄存器地址 + RESTART + 设备地址/R + 读数据 + STOP。

6. 类 API 参考

6.1 构造函数 & 初始化

构造函数
SC8815::SC8815(uint8_t i2c_addr = 0x74, TwoWire *wire = &Wire);

创建 SC8815 实例。i2c_addr 为 7-bit I²C 设备地址，wire 为使用的 I²C 总线指针。仅保存参数，不执行任何 I/O 操作。

析构函数
SC8815::~SC8815();

默认析构，无特殊清理。

begin — 设备检测
bool begin();

检测 I²C 总线上是否存在 SC8815 设备（发送地址 + 检查 ACK）。

返回：true = 设备响应 ACK；false = 通信失败。

警告：不会调用 Wire.begin()。调用前必须已自行初始化 Wire。

6.2 底层 I²C 操作

writeRegister — 写单字节
bool writeRegister(uint8_t reg, uint8_t value);

写单字节到指定寄存器。返回 true 表示写入成功（收到 ACK）。

readRegister — 读单字节
bool readRegister(uint8_t reg, uint8_t *value);

从指定寄存器读单字节。value 输出读取值。返回 true 表示读取成功。

6.3 VBAT_SET — 电池充电配置

0x00 RW setVBATSet
bool setVBATSet(SC8815_IRCOMP_t ircomp, SC8815_VBAT_SEL_t sel, SC8815_CSEL_t csel, SC8815_VCELL_t vcell);

设置电池充电目标电压及相关配置。

参数说明
ircomp电池内阻补偿 (IRCOMP)
selVBAT 设定方式 (VBAT_SEL)
csel电池串联节数 (CSEL)，仅 sel=INTERNAL 有效
vcell单节目标电压 (VCELL)，仅 sel=INTERNAL 有效

位布局：

7IRCOMP[1] 6IRCOMP[0] 5VBAT_SEL 4CSEL[1] 3CSEL[0] 2VCELL[2] 1VCELL[1] 0VCELL[0]
在放电模式下此寄存器配置不影响输出，但仍建议正确设置。
0x00 RW getVBATSet
bool getVBATSet(uint8_t *reg_value);

读取 VBAT_SET 寄存器原始值。

6.4 VBUSREF_I — 内部 VBUS 参考电压 0x01+0x02

RW setVBUSRefI
bool setVBUSRefI(uint16_t ref_mV);

设置内部 VBUS 参考电压 (10-bit)。

范围：2~2048 mV。

公式：VBUSREF_I = (4 × VHI + VLO + 1) × 2mV

输出：VBUS = VBUSREF_I × VBUS_RATIO

仅在 FB_SEL=0 (内部设定) 时生效。写入前请确认 CTRL1 的 FB_SEL=0。
RW getVBUSRefI
bool getVBUSRefI(uint16_t *ref_mV);

读取当前内部 VBUS 参考电压 (mV)。

6.5 VBUSREF_E — 外部 FB 参考电压 0x03+0x04

RW setVBUSRefE
bool setVBUSRefE(uint16_t ref_mV);

设置外部 FB 引脚参考电压 (10-bit)。

推荐范围：0.5V ~ 2.048V。

输出：VBUS = VBUSREF_E × (1 + RUP / RDOWN)

仅在 FB_SEL=1 (外部电阻设定) 时生效。FB_SEL=1 时仍需配置 VBUS_RATIO（仅影响 ADC 监控量程）。
RW getVBUSRefE
bool getVBUSRefE(uint16_t *ref_mV);

读取当前外部 FB 参考电压 (mV)。

6.6 IBUS_LIM — IBUS 电流限制 0x05

RW setIBUSLimit
bool setIBUSLimit(uint16_t current_mA);

设置 IBUS 电流限制（充/放电模式均生效）。

公式：IBUS_LIM(A) = (IBUS_LIM_SET + 1) / 256 × IBUS_RATIO × 10mΩ / RS1

自动读取当前 IBUS_RATIO 和 RS1=10mΩ 计算寄存器值。

实际限流精度取决于采样电阻温度系数。最小值不低于 0.3A（数据手册要求）。
RW getIBUSLimit
bool getIBUSLimit(uint16_t *current_mA);

读取当前 IBUS 电流限制 (mA)。

6.7 IBAT_LIM — IBAT 电流限制 0x06

RW setIBATLimit
bool setIBATLimit(uint16_t current_mA);

设置 IBAT 电流限制（充/放电模式均生效）。

公式：IBAT_LIM(A) = (IBAT_LIM_SET + 1) / 256 × IBAT_RATIO × 10mΩ / RS2

最小值不低于 0.3A（数据手册要求）。
RW getIBATLimit
bool getIBATLimit(uint16_t *current_mA);

读取当前 IBAT 电流限制 (mA)。

6.8 VINREG — 动态电源管理 0x07

7. 使用示例

7.1 充电模式

#include <Wire.h>
#include "sc8815.h"

SC8815 sc;

void setup() {
    Serial.begin(115200);
    Wire.begin(21, 22);
    Wire.setClock(400000);

    if (!sc.begin()) {
        Serial.println("SC8815 未检测到!");
        while (1);
    }

    // === 充电配置 ===
    // 3S 电池, 4.20V/节 充电
    sc.setChargeVoltage(3, SC8815_VCELL_4V20);

    // IBUS=2A, IBAT=2A
    sc.setChargeCurrent(2000, 2000);

    // 充电模式: 内部 FB, 启用涓流, 启用终止, 启用 OVP
    sc.setCTRL1(SC8815_ICHAR_SEL_IBUS,
                SC8815_DIS_TRICKLE_ENABLE,
                SC8815_DIS_TERM_ENABLE,
                SC8815_FB_SEL_INTERNAL,
                SC8815_TRICKLE_SET_70,
                SC8815_DIS_OVP_DISABLE);   // DIS_OVP_ENABLE=0 → OVP 保护开启

    // 启动 ADC
    sc.setCTRL3(SC8815_EN_PGATE_OFF,
                SC8815_GPO_CTRL_OD,
                SC8815_AD_START_ENABLE,
                SC8815_ILIM_BW_SEL_5K,
                SC8815_LOOP_SET_NORMAL,
                SC8815_DIS_SHORT_FOLDBACK_ENABLE,
                SC8815_EOC_SET_1_10,
                SC8815_EN_PFM_DISABLE);

    // 确认在充电模式
    sc.enableChargingMode();

    Serial.println("充电模式已就绪");
}

void loop() {
    uint16_t vbus, vbat, ibus, ibat;
    sc.readVBUSVoltage(&vbus);
    sc.readVBATVoltage(&vbat);
    sc.readIBUSCurrent(&ibus);
    sc.readIBATCurrent(&ibat);

    Serial.printf("VBUS=%umV  VBAT=%umV  IBUS=%umA  IBAT=%umA\n",
                  vbus, vbat, ibus, ibat);

    if (sc.isACOK()) Serial.println("  → 适配器已插入");
    if (sc.isEOC())  Serial.println("  → 充电已终止");

    delay(1000);
}

7.3 中断监控

void checkFaults() {
    uint8_t status;
    sc.getStatus(&status);
    Serial.printf("STATUS=0x%02X\n", status);

    if (sc.isOTP())        Serial.println("⚠ 过温保护! (Tj > 165°C)");
    if (sc.isVBUSShort())  Serial.println("⚠ VBUS 短路! (VBUS < 1V)");
    if (sc.isACOK())       Serial.println("✓ 适配器已插入");
    if (sc.isINDET())      Serial.println("✓ 检测到负载插入");
    if (sc.isEOC())        Serial.println("✓ 充电已终止");
}

void maskInterrupts() {
    // 屏蔽 OTP 和 VBUS_SHORT 中断 (仅屏蔽, 不影响保护动作)
    sc.setMask(false,   // AC_OK: 不屏蔽
               false,   // INDET: 不屏蔽
               true,    // VBUS_SHORT: 屏蔽
               true,    // OTP: 屏蔽
               false);  // EOC: 不屏蔽
}

7.5 外部 FB 电阻分压放电

// 外部 RUP=100kΩ, RDOWN=10kΩ → 分压比 = 1 + 100/10 = 11
// VBUS 目标 12V → VBUSREF_E = 12V / 11 ≈ 1091mV
void setupExternalFB() {
    sc.setCTRL1(SC8815_ICHAR_SEL_IBUS,
                SC8815_DIS_TRICKLE_ENABLE,
                SC8815_DIS_TERM_ENABLE,
                SC8815_FB_SEL_EXTERNAL,      // 外部 FB 方式
                SC8815_TRICKLE_SET_70,
                SC8815_DIS_OVP_DISABLE);

    sc.setVBUSRefE(1091);  // 1091mV 参考电压

    // VBUS_RATIO 仍需正确设置以配合 ADC 监控
    sc.setRatio(SC8815_IBUS_RATIO_3x,
                SC8815_IBAT_RATIO_12x,
                SC8815_VBUS_RATIO_12_5x,     // 12V → 12.5x 比率
                SC8815_VBAT_MON_RATIO_12_5x);

    // 切换到放电模式
    sc.setCTRL0(SC8815_EN_OTG_ENABLE,
                SC8815_VINREG_RATIO_100x,
                SC8815_FREQ_300kHz,
                SC8815_DT_40ns);
}

8. 注意事项

8.1 寄存器操作

  • Read-Modify-Write：所有 setCTRL0()、setCTRL1()、setCTRL2()、setRatio()、setMask() 均采用 RMW 策略保护寄存器中的 Reserved bits，避免因直接覆写导致芯片异常。
  • CTRL2 FACTORY 位：数据手册强制要求 MCU 上电后将 CTRL2[3] 置 1。库中 setCTRL2() 默认参数已自动设为 SC8815_FACTORY_POWERUP。
  • PSTOP 限制：CTRL0、CTRL1、CTRL2、CTRL3 寄存器建议仅在 PSTOP=HIGH (待机状态) 时修改。

8.2 I²C 初始化

  • begin() 不会调用 Wire.begin()。调用方必须在调用 begin() 之前自行完成 I²C 总线的引脚配置和时钟设置。
  • 推荐 I²C 速率 400kHz (Fast Mode)。

8.3 ADC 使用

  • 读取任何 ADC 值之前，必须先通过 setCTRL3() 将 ad_start 设为 SC8815_AD_START_ENABLE。
  • ADC 为 10-bit 分辨率，分布在两个寄存器中 (高 8 位 + 低 2 位)。
  • VBUS_RATIO=12.5x 时最大可测 ~25.6V，超出将饱和。

8.4 电流限制假设

  • 所有电流计算默认假设采样电阻 RSNS1 = RSNS2 = 10mΩ。如果实际电路使用不同阻值的采样电阻，需要修改库代码中的对应常量。
  • 实际限流精度取决于采样电阻的温度系数。

8.5 保护功能

保护触发条件行为
UVP输入电压低于阈值停止充电 / 关断输出
OVP输出电压超过阈值停止充电 (可通过 DIS_OVP 控制)
OCP电流超过 IBUS_LIM / IBAT_LIM限流
SCPVBUS < 1V (短路)电流折返至 22%(IBUS) / 10%(IBAT)
OTPTj > 165°C关断输出, 滞后 ~35°C 恢复

8.6 OVP 命名约定

库中 DIS_OVP 等以 DIS_ 开头的枚举，其 _ENABLE 值对应 bit = 0 (即"启用"该禁用位 = 保护开启)，_DISABLE 值对应 bit = 1 (即该位生效 = 保护关闭)。请仔细阅读每个枚举的注释以免配置错误。
